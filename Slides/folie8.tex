Der Robinson--Kalk"ul
\end{center}
\vspace*{0.5cm}

\footnotesize
\textbf{Vereinbarung:} Bei Klauseln Allquantoren weggelassen.

\textbf{Beispiel}: Wir schreiben \\[0.3cm]
\hspace*{1.3cm} $\textsl{pop}(\textsl{push}(x_1,s_1))$ \\[0.3cm]
statt \\[0.3cm]
\hspace*{1.3cm} $\forall x_1\in \mathbb{N}: \forall s_1\in\textsl{Stack}:\textsl{pop}(\textsl{push}(x_1,s_1))$. 
\vspace*{0.5cm}

\textbf{Vereinbarung}: Klauseln sind Mengen von Literalen \\[0.3cm]
\hspace*{1.3cm} $l_1 \vee \cdots \vee l_n \;=\; \{l_1, \cdots, l_n\}$

\textbf{Definition}: Schlu\3--Regel ist Paar der Form \\[0.3cm]
\hspace*{1.3cm} $\langle \{k_1, \cdots, k_n\}, k \rangle$ \\[0.3cm]
 $k$ und $k_i$: pr"adikatenlogische Klauseln.

$k_1$, $\cdots$, $k_n$: \ \ Pr"amissen

$k$: \quad \quad \quad \quad Konklusion


\textbf{Schreibweise}: \\[0.1cm]
\hspace*{1.3cm} $\schluss{k_1 \quad \cdots \quad k_n}{k}$ \\[0.1cm]

\textbf{Definition}: Substitutions--Regel \\[0.3cm]
\hspace*{1.3cm} $\schluss{k}{k\tau}(\emph{Subst})$ \\[0.7cm]
$k$: \quad Klausel

$\tau$: \quad Substitution

\footnotesize
\textbf{Beobachtung}:  Klauseln ``schrumpfen'' durch Subst-Regel.

\textbf{Betrachte}:
\begin{enumerate}
\item $k = q(x,d) \vee q(c,y)$
\item $\tau = [ x \mapsto c, y \mapsto d ]$
\end{enumerate}
Dann gilt: 
$$
\begin{array}{lll}
 k\tau & = & q(c,d) \vee q(c,d)   \\
       & = & \{ q(c,d), q(c,d) \} \\
       & = & \{ q(c,d) \}         \\
       & = & q(c,d) \\
\end{array}
$$
Also:
\hspace*{1.3cm} $\schluss{q(x,d) \vee q(c,y)}{q(c,d)}(\textsl{Subst})$.
\vspace{0.3cm}

\textbf{Definition}: (Schnitt--Regel)  Es sei
\begin{enumerate}
\item $k_1$, $k_2$: \ prädikatenlogische Klauseln
\item $l$: \quad \quad \ prädikatenlogisches Literal
\end{enumerate}
\hspace*{1.3cm} $\schluss{k_1 \vee l \quad\quad \komplement{\,l\,} \vee k_2}{k_1 \vee k_2}(\textsl{Schnitt})$ 
\vspace{0.3cm}

\textbf{Definition}: Kalkül = Menge von Schluss--Regeln
\\[0.3cm]
\hspace*{1.3cm} $\textsl{Simple} := \{ \textsl{Subst}, \textsl{Schnitt} \}$

Herleitungs--Begriff
\end{center}
\vspace{0.5cm}

\footnotesize
\textbf{Definition}: $M \vdash_\mathcal{K} k$
\begin{enumerate}
\item Falls $k \in M$, dann $M \vdash_\mathcal{K} k$.

\item Falls 
     \begin{enumerate}
     \item  $M \vdash_\mathcal{K} k_i$  für alle $i=1,\cdots,n$ und
     \item $\schluss{k_1 \quad \cdots \quad k_n}{k}$ Schluss aus $\mathcal{K}$,
     \end{enumerate}
     dann $M \vdash_\mathcal{K} k$.
\end{enumerate}
Sprechweise: ``{\em $k$ kann mit $\mathcal{K}$ aus $M$ hergeleitet werden}''.

\textbf{Beispiel}: Signatur
\begin{enumerate}
\item $\mathbb{T} := \{ \mathbb{B}, \mathbb{N} \}$.
\item $\textsl{Fz} := \{ s \}$ und $s: \mathbb{N} \rightarrow \mathbb{N}$.
\item $\textsl{Pz} := \{ < \}$ und $<: \mathbb{N} \times \mathbb{N} \rightarrow \mathbb{B}$.
\item $\mathcal{V} := \{ x, y, z, u,v,w \}$.
\end{enumerate}
Klauseln:
\begin{enumerate}
\item $g_1 := \neg\; x \!<\! x$.
\item $g_2 := \neg\; x \!<\! y \;\vee\; \neg\; y \!<\! z \;\vee\; x \!<\! z$.
\item $g_3 := x \!<\! s(x)$.
\item $g_4 := x \!<\! s(s(x))$.
\end{enumerate}

\footnotesize
Setze: $M := \{ g_1, g_2, g_3 \}$.  

Zeige: $M \vdash_\textsl{Simple} g_4$.
\begin{enumerate}
\item Aus $g_2$ folgt mit $\tau = [y \mapsto s(x)]$

      $$\schluss{\neg\; x \!<\! y \;\vee\; \neg\; y \!<\! z \;\vee\; x \!<\! z}{\neg\; x \!<\! s(x) \;\vee\; \neg\; s(x) \!<\! z \;\vee\; x \!<\! z}(\textsl{Subst})$$
\item Aus $g_3$ und der letzten Formel folgt mit Schnitt

      $$\schluss{x \!<\! s(x) \quad\quad \neg\; x \!<\! s(x) \;\vee\; \neg\; s(x) \!<\! z \;\vee\; x \!<\! z}{\neg\; s(x) \!<\! z \;\vee\; x \!<\! z}$$
\item Daraus folgt mit $\tau = [z \mapsto s(s(x))]$

      $$\schluss{\neg\; s(x) \!<\! z \;\vee\; x \!<\! z}{\neg\; s(x) \!<\! s(s(x)) \;\vee\; x \!<\! s(s(x))}(\textsl{Subst})$$
\item Aus $g_3$ folgt mit $\tau = [ x \mapsto s(x) ]$

      $$\schluss{x \!<\! s(x)}{s(x) \!<\! s(s(x))}(\textsl{Subst})$$
\item Aus den letzten beiden Formeln folgt mit Schnitt

      $$\schluss{s(x) \!<\! s(s(x)) \quad\quad \neg\; s(x) \!<\! s(s(x)) \;\vee\; x \!<\! s(s(x))}{x \!<\! s(s(x))}$$ 
\end{enumerate}

\begin{slide}{}
\normalsize
\begin{center}
Resolutions--Regel (Motivation)
\end{center}
\vspace{0.5cm}

\footnotesize
\textbf{Frage}: \quad \ Wie wähle ich Substitution $\tau$ in $(\textsl{Subst})$?

\textbf{Antwort}: \ So, dass Schnitt--Regel anwendbar wird!
$$
  \schluss{\hspace*{-1cm} \schluss{k_1 \vee l_1}{k_1\tau_1 \vee l_1\tau_1}(\textsl{Subst}) \quad  
            \schluss{l_2 \vee k_2}{l_2\tau_2 \vee k_2\tau_2}(\textsl{Subst})\hspace*{-1cm} }{k_1\tau_1 \vee k_2\tau_2}(\textsl{Schnitt})
$$
\textbf{Forderung}:
$$ l_1\tau_1 = \komplement{\,l_2\,}\tau_2 $$

\textbf{Beispiel}:
\begin{enumerate}
\item $k_1 \vee l_1 \;=\; x \!<\! s(x)$
\item $l_2 \vee k_2 \;=\; \neg\; s(x) \!<\! z \;\vee\; x \!<\! z$
\item $(x < s(x))\tau_1 = (x < z)\tau_2$
\item $\tau_1 = [x \mapsto s(x)]$
\item $\tau_2 = [z \mapsto s(s(x))]$ 
\item $k_1\tau_1 \vee k_2\tau_2 \;=\; x \!<\! s(s(x))$
\end{enumerate}

\textbf{Definition}: $\pi = [x_1 \mapsto y_1, \cdots, x_n \mapsto y_n]$ \\[0.3cm]
ist  \emph{Variablen--Umbenennung} falls 
\begin{enumerate}
\item $\{y_1, \cdots, y_n\} \cap \{x_1,\cdots, x_n\} = \emptyset$. 
\item $i \not= j \Rightarrow y_i \not= y_j$,
\end{enumerate}

Resolutions--Regel
\end{center}
\vspace{0.5cm}

\footnotesize
\textbf{Definition}: (Resolutions--Regel) \quad   Es gelte:
    \begin{enumerate}
    \item $k_1$, $k_2$: prädikatenlogische Klauseln,
    \item $l_1$, $l_2$: prädikatenlogische Literale,
    \item $\pi$: Variablen--Umbenennung,
    \item $l_1 \doteq \komplement{\,l_2}\pi$: lösbare syntaktische Gleichung,
    \item $\mu = \textsl{mgu}(l_1, \komplement{\,l_2\,})$.
    \end{enumerate}
     Dann: \emph{Resolutions--Regel} \\[0.5cm]
     \hspace*{1.3cm} $\schluss{k_1 \vee l_1 \quad\quad l_2 \vee k_2}{k_1\mu \vee k_2\pi\mu}(\textsl{Res})$ \\[0.2cm]

\textbf{Bemerkung}: $\textsl{Res} = 2 * (\textsl{Subst}) + (\textsl{Schnitt})$

$$
  \schluss{\hspace*{-1cm} { \; \atop \schluss{k_1 \vee l_1}{k_1\mu \vee l_1\mu}(\textsl{Subst})} \quad 
            \schluss{\hspace*{-1cm}l_2 \vee k_2\hspace*{-1cm}}{\hspace*{-1cm}\schluss{l_2\pi \vee k_2\pi}{l_2\pi\mu \vee
                k_2\pi\mu}(\textsl{Subst})\hspace*{-3cm}}(\textsl{Subst}) \hspace*{-1cm}}{k_1\mu \vee k_2\pi\mu}(\textsl{Schnitt})
$$

\begin{center}
Robinson--Kalkül
\end{center}
\vspace{0.5cm}

\footnotesize
\textbf{Definition}: (Faktorisierungs--Regel) 
\begin{enumerate}
\item  $k$: prädikatenlogische Klausel,
\item  $l_1$ und $l_2$: prädikatenlogische Literale,
\item  $l_1 \doteq l_2$: lösbare syntaktische Gleichung
\item  $\mu = \textsl{mgu}(l_1, l_2)$
\end{enumerate}
\hspace*{1.3cm} $\schluss{k \vee l_1 \vee l_2}{k\mu \vee l_1\mu}(\textsl{Fakt})$ \\[0.1cm]

\textbf{Beispiel}:

 \hspace*{1.3cm} $\schluss{q(x,d) \vee q(c,y)}{q(c,d)}(\textsl{Fakt})$.

\textbf{Bemerkung}: $(\textsl{Fakt})$ ist Spezialfall von $(\textsl{Subst})$

\textbf{Definition}: Robinson--Kalkül \\[0.3cm]
\hspace*{1.3cm} $\mathcal{R} := \{ \textsl{Res}, \textsl{Fakt} \}$

\textbf{Korrektheits--Satz}:

Falls $M \vdash_\mathcal{R} k$, dann $M \models k$.
\vspace{0.5cm}

\textbf{Widerlegungs--Vollständigkeit}:

Falls $M \models \falsum$, dann $M \vdash_\mathcal{R} \falsum$.

  \end{enumerate}
\end{enumerate}

Strukturen $\textsl{S}$

      \vspace{0.3cm}

      $\schluss{\textsl{rot}(k(d)) \quad \neg \textsl{rot}(x) \vee \textsl{fliegt}(x)}{\textsl{fliegt}(k(d))}(\mathcal{R})$
\item Daraus folgt mit $k_2$
      \vspace{0.3cm}

      $\schluss{\textsl{fliegt}(k(d)) \quad \neg \textsl{fliegt}(k(x)) \vee \textsl{glück}(x)}{\textsl{glück}(d)}(\mathcal{R})$
\item Daraus folgt mit $k_6$
      \vspace{0.3cm}

      $\schluss{\textsl{glück}(d) \quad  \neg \textsl{glück}(d)}{\falsum}(\mathcal{R})$
\end{enumerate}
