\item \emph{Konstanten}: Namen für feste Objekte
      \begin{enumerate}
      \item $\pi$ mit $\pi = 3.14159\cdots$
      \item \texttt{Adam}, \texttt{Eva}
      \end{enumerate}

\item \emph{Funktions-Zeichen}
      \begin{enumerate}
      \item $+$: 2-stelliges Funktions-Zeichen

      \item $\sqrt{\ }$: 1-stelliges Funktions-Zeichen

      \item Allgemein: sei \\[0.3cm]

            \hspace*{1.3cm} $\bullet$ $t_1$, $\cdots$, $t_n$ Terme \\[0.3cm]
            Dann: $f(t_1,\cdots,t_n)$ neuer \emph{Term}

            \hspace*{1.3cm} $\bullet$ \texttt{vater}, \texttt{mutter}: 1-stellige Funktions-Zeichen \\[0.3cm]

\begin{center}
Syntax von Formeln
\end{center}
\vspace{0.5cm}

\footnotesize
\begin{enumerate}
\setcounter{enumi}{4}
\item \emph{Junktoren} verknüpfen Formeln

   \begin{enumerate}
   \item Konjunktion: \emph{und} \quad ``$\wedge$'' \\[0.3cm]
     \hspace*{1.3cm} $2 < 7$ und $7 < 10$ \\[0.3cm] 
     \hspace*{1.3cm} $2 < 7 \wedge 7 < 10$

   \item Disjunktion: \emph{oder} \quad ``$\vee$'' \\[0.3cm]
     \hspace*{1.3cm} $x \leq y$ oder $y \leq x$. \\[0.3cm] 
     \hspace*{1.3cm} $x \leq y \vee y \leq x$.

   \item Negation: \emph{nicht} \quad ``$\neg$'' \\[0.3cm]
     \hspace*{1.3cm}  $x^2$ ist nicht 2\\[0.3cm]
     \hspace*{1.3cm} $\neg\; x^2 = 2$.

   \item Implikation: \emph{wenn, dann} \quad  ``$\rightarrow$'' \\[0.3cm]
     \hspace*{1.3cm} wenn $x < y$, dann $x^2 < y^2$. \\[0.3cm]
     \hspace*{1.3cm} $x < y \rightarrow x^2 < y^2$
   \end{enumerate}
\item \emph{Quantoren} klären Verwendung von Variablen
  \begin{enumerate}
  \item All-Quantor: \quad ``$\forall$'' \\[0.3cm]
\hspace*{1.3cm} Für alle ganzen Zahlen $z$ gilt: $z^2 \geq 0$ \\[0.3cm]
\hspace*{1.3cm} $\forall z \in \mathbb{Z} : z^2 \geq 0$

\item Existenz-Quantor: \quad ``$\exists$'' \\[0.3cm]
\hspace*{1.3cm} Für alle  $u, v \in \mathbb{R}$ gibt es $w$ aus $\mathbb{R}$ mit 
                $u + w = v$. \\[0.3cm]
\hspace*{1.3cm} $\forall u,v \in \mathbb{Z}:  \exists w \in \mathbb{Z} : u + w = v$
  \end{enumerate}
\end{enumerate}

\item Ist folgende Formel richtig oder falsch? \\[0.3cm]

      Problem: Wie soll Formel gelesen werden
      \begin{enumerate}
      \item $(1 < 2 \vee 2 = 2) \wedge 0 = 1$: falsch
      \item $1 < 2 \vee (2 = 2 \wedge 0 = 1)$: richtig
      \end{enumerate}
\item Konsequenz: Formeln klammern!
\item Bindungsregeln zur Vereinfachung \\[0.5cm]  
      In \textbf{Arithmetik}: Punkt vor Strich \\[0.3cm]
\hspace*{1.3cm} 
      $x + y \cdot z$ wird gelesen als $x + (y \cdot z)$
\\[0.6cm]
      In \textbf{Logik}: folgende Konvention
  \begin{enumerate}
  \item $\neg$ bindet am stärksten
  \item $\wedge$ und $\vee$ binden gleichstark
  \item $\rightarrow$ bindet schwächer als $\wedge$, $\vee$
  \item $\leftrightarrow$ bindet am schwächsten
  \end{enumerate}
  Beispiel: \\[0.3cm]
      \hspace*{1.3cm} $P \wedge Q \rightarrow R \leftrightarrow \neg R \rightarrow \neg P \vee \neg Q$ \\[0.3cm]
wird gelesen als \\[0.3cm]
\hspace*{1.3cm} 
      $\bigl((P \wedge Q) \rightarrow R\bigr) \leftrightarrow \Bigl(\,(\neg R) \rightarrow \bigl((\neg P) \vee (\neg Q)\bigr)\,\Bigr)$ 
\end{enumerate}

\begin{center}
Naive Mengenlehre
\end{center}
\vspace{0.5cm}

\footnotesize
Frage:   Was sind Mengen? \\
Antwort: Ansammlung verschiedener Objekte. 

Beispiele:
\begin{enumerate}
\item Menge der Zahlen 1, 2 und 5: \quad $\{1, 2, 5 \}$.
\item Menge der Buchstaben:       \quad $\{a, b, c, \cdots, x, y, z \}$.
\item Menge der natürlichen Zahlen: $\mathbb{N} = \{ 0, 1, 2, 3, \cdots \}$.
\end{enumerate}
Notation:  Statt \\[0.3cm]
\hspace*{1.3cm} ``Die Zahl $2$ ist eine Element der Menge $\{1, 2, 3\}$'' \\[0.3cm]
schreiben wir:  $2 \in \{ 1, 2, 3 \}$.

\begin{enumerate}
\item Reihenfolge spielt keine Rolle.\\[0.1cm]
      \hspace*{1.3cm} $\{ 1, 2, 3 \} = \{ 3, 2, 1 \}$
\item Eine Menge kann jedes Element höchstens einmal enthalten: \\[0.1cm]
\hspace*{1.3cm} $\{ 1, 2, 2, 3 \} = \{ 1, 1, 2, 3, 3 \}$.
\end{enumerate}

\begin{center}
\framebox{\framebox{
  \begin{minipage}[c]{10cm}
{\bf Extensionalitäts-Prinzip}: \\
Zwei Mengen sind gleich, wenn sie die selben Elemente enthalten:
  \end{minipage}
}}  
\end{center}
Formalisiert:
$$ M = N \leftrightarrow (\forall x: x \in M \leftrightarrow x \in N) $$

\begin{center}
Komprehensions-Axiom
\end{center}
\vspace{0.5cm}

\footnotesize
\begin{enumerate}
\item Frage: Wie definieren wir Mengen?
\item Erste Antwort (Cantor): Komprehensions-Axiom

Sei $p(x)$ Formel, in der $x$ auftritt \\[0.3cm]
\hspace*{1.3cm} $M = \{ x \;|\; p(x) \}$ \\[0.3cm]
bezeichnet Menge aller Objekte, für die $p$ gilt.

Lesart:  $M$ ist die Menge aller $x$, für die $p(x)$ gilt.

Beispiel:
\begin{enumerate}
\item Sei $p(x) := (\exists y \in \mathbb{N}: x = y^2)$.
\item $Q = \{ x \mid p(x) \} =  \Bigl\{ x \mid \exists y \in \mathbb{N}: x = y^2 \Bigr\}$

      Menge der Quadrat-Zahlen
\item Russell-Menge \\[0.3cm]
      \hspace*{1.3cm} 
      $R = \{ x \mid \neg x \in x \}$

      führt zu Parodoxie
\item  Ausweg: $R$ nicht wohldefiniert

       Komprehensions-Axiom nicht zuläßig
\end{enumerate}
\item Heutige Antwort: 

      Einschränkung des Komprehensions-Axioms
\end{enumerate}

  Vereinigungs-Menge
\end{center}
\footnotesize

\textbf{Gegeben}: Zwei Mengen $M_1$, $M_2$ \\[0.3cm]
\textbf{Definition}: \\[0.3cm]
\hspace*{1.3cm} $M_1 \cup M_2 = \{ x  \is x \in M_1 \vee x \in M_2 \}$  \\[0.3cm]
heißt Vereinigungs-Menge.

Beispiel:  $\{1, 3, 5, 6, 7 \} \cup \{4, 5, 6 \} = \{ 1, 3, 4, 5, 6, 7 \}$

\normalsize
\begin{center}
Mengen-Komplement
\end{center}
\vspace{0.5cm}

\footnotesize
Gegeben: Mengen $M_1$ und $M_2$. \\[0.3cm]
Definition: $M_1 \backslash M_2 = \{ x \mid  x \in M_1 \wedge x \not\in M_2 \}$

Beispiel:  $\{1, 3, 5, 6, 7 \} \backslash \{4, 5, 6 \} = \{ 1, 3, 7 \}$

\rule{17cm}{0.5mm}

\textbf{Satz}: Für beliebige Mengen $M$ und $N$ gilt:
\begin{enumerate}
\item $M \cap N \subseteq M$
\item $M \subseteq M \cup N$
\item $M \backslash N \subseteq M$
\item $M \backslash (M \backslash N) = M \cap N$
\item $N \backslash (M \backslash N) = N$
\end{enumerate}

\normalsize
\begin{center}
    Mengen-Algebra
\end{center}
\footnotesize
\begin{enumerate}
\item Idempotenz:\quad      $M \cup M = M$,     $M \cap M = M$
\item Neutrales Element:    $M \cup \emptyset = M$, $M \cap \emptyset = \emptyset$

\item Kommutativ-Gesetze

        $M_1 \cup M_2 = M_2 \cup M_1$

        $M_1 \cap M_2 = M_2 \cap M_1$
\item Assoziativ-Gesetze

      $(M_1 \cup M_2) \cup M_3 = M_1 \cup (M_2 \cup M_3)$

      $(M_1 \cap M_2) \cap M_3 = M_1 \cap (M_2 \cap M_3)$

\item Distributiv-Gesetze

      $(M_1 \cup M_2) \cap N = (M_1 \cap N) \cup (M_2 \cap N)$

      $(M_1 \cap M_2) \cup N = (M_1 \cup N) \cap (M_2 \cup N)$
\item DeMorgan-Gesetze

      $N \backslash (M_1 \cup M_2) = (N \backslash M_1) \cap (N \backslash M_2)$

      $N \backslash (M_1 \cap M_2) = (N \backslash M_1) \cup (N \backslash M_2)$
\end{enumerate}

Verallgemeinertes kartesisches Produkt
\end{center}

\footnotesize

\hspace*{6.2cm} $ = M_1 \times M_2 \times M_3$.

\begin{center}
    \fbox{\fbox{verallgemeinertes kartesisches Produkt}}
\end{center}

\normalsize
\begin{center}
    Binäre Relation
\end{center}
\footnotesize
Gilt $R \subseteq M \times M$, so heißt $R$ eine \emph{binäre Relation} auf $M$.

Beispiele:
\begin{enumerate}
\item $\{ \langle x, x \rangle \in \mathbb{N} \times \mathbb{N} \is x \in \mathbb{N} \}$.
\item $\{ \langle x, y \rangle \in \mathbb{N} \times \mathbb{N} \is x < y \}$.
\item $\{ \langle x, y \rangle \in \mathbb{N} \times \mathbb{N} \is y = x * x \}$.
\end{enumerate}

Schreibweise: \quad $M^2 := M \times M$ \\
Allgemein wird $M^n$ per Induktion nach $n$ definiert:
\begin{enumerate}
\item $M^1 := M$ 
\item $M^{n+1} := M^n \times M$
\end{enumerate}

\normalsize
\begin{center}
  Ordnungs-Relationen
\end{center}

\footnotesize
\textbf{Infix-Schreibweise}: Schreibe $x R y$ statt $\langle x, y \rangle \in R$.

\textbf{Definition}:  $R \subseteq M^2$ ist \emph{reflexiv} g.d.w. \\[0.3cm]
\hspace*{1.3cm} $\forall x \in M: x R x$.

\textbf{Definition}:  $R \subseteq M^2$ ist \emph{symmetrisch} g.d.w. \\[0.3cm]
\hspace*{1.3cm}  $\forall x, y \in M: x R y \rightarrow  y R x$. 

\textbf{Definition}:  $R \subseteq M^2$ ist \emph{anti-symmetrisch} g.d.w. \\[0.3cm]
\hspace*{1.3cm}  $\forall x, y \in M: x R y \wedge y R x \rightarrow x = y$. 

\textbf{Definition}:  $R \subseteq M^2$ ist \emph{transitiv} g.d.w. \\[0.3cm]
\hspace*{0.3cm} $\forall x, y, z \in M: x R y \wedge y R z \rightarrow x R z$. 

\textbf{Definition}:  $R \subseteq M^2$ ist \emph{total} g.d.w. \\[0.3cm]
\hspace*{0.3cm} $\forall x, y\in M: x R y \vee y R x$. 

\textbf{Definition}: \\
 $R \subseteq M^2$  \emph{partielle Ordnung} im Sinne von $\leq$ g.d.w. 
\begin{enumerate}
\item $R$ ist reflexiv.
\item $R$ ist anti-symmetrisch. 
\item $R$ ist transitiv.
\end{enumerate}

\textbf{Definition}: 
 $R \subseteq M^2$ ist eine \emph{totale Ordnung} \\
 (auch: \emph{lineare Ordnung}) im Sinne von $\leq$ g.d.w. 
\begin{enumerate}
\item $R$ ist partielle Ordnung.
\item $R$ ist total. 
\end{enumerate}

Aufgaben
\end{center}
\vspace{0.5cm}

\footnotesize
\textbf{Aufgabe 1}: Wie sieht die kleinste Ordnungs-Relation im Sinne von $\leq$ auf 
$S = \{1, 2, 3\}$ aus?
\vspace{0.5cm}

\textbf{Definition}: Sei $R \subseteq M \times M$ eine Ordnungs-Relation im Sinne von
$\leq$.  Die Ordnung $R$ heißt \underline{\emph{total}} g.d.w. \\[0.3cm]
\hspace*{1.3cm} $\forall x,y \in M: x R y \vee y R x$.

\textbf{Beispiel}: Die Relation $\{ \langle x, y \rangle \in \mathbb{N}^2 \is x \leq y \}$
ist total.

\textbf{Gegenbeispiel}: Die Relation \\[0.3cm]
\hspace*{1.3cm} $\mathtt{Teiler} = \{ \langle x, y \rangle \in \mathbb{N}^2 \is \exists z \in \mathbb{N}: x * z = y \}$ \\[0.3cm]
ist \underline{nicht} total.
\vspace{0.5cm}

\textbf{Aufgabe 2}: Geben Sie eine \underline{totale} Ordnungs-Relation im Sinne von $\leq$ auf 
$S = \{1, 2, 3\}$ an.
\vspace{0.5cm}

\textbf{Definition}: Sei $R_1 \subseteq K \times M$ und $R_2 \subseteq M \times N$
$$  R_2 \circ R_1 = \{ \langle x, z \rangle \in K \times N \is \exists y \in M: \langle x, y \rangle \in R_1\; \wedge
                                                                                \langle y, z \rangle \in R_2 \}$$

\textbf{Aufgabe 3}: Seien $R_1, R_2 \subseteq M \times M$. \\[0.3cm]
Beweisen oder widerlegen Sie:
\begin{enumerate}
\item Falls $R_1$ und $R_2$ reflexiv sind, \\[0.3cm]
      dann ist auch $R_1 \circ R_2$ reflexiv.
\item Falls $R_1$ und $R_2$ symmetrisch sind, \\[0.3cm]
      dann ist auch $R_1 \circ R_2$ symmetrisch.
\item Falls $R_1$ und $R_2$ transitiv sind, \\[0.3cm]
      dann ist auch $R_1 \circ R_2$ transitiv.
\end{enumerate}

\begin{center}
Potenz-Menge
\end{center}
\vspace{0.5cm}

\footnotesize
\textbf{Schreibweise}: Sei $f: M \rightarrow N$ Funktion.  \\[0.3cm]
\hspace*{1.3cm} $\{ f(x) \is x \in M \} := \{ y \in N \is \exists x \in M: y = f(x) \}$ \\[0.3cm]
\hspace*{3.9cm} $f(M) := \{ y \in N \is \exists x \in M: y = f(x) \}$

\footnotesize
\textbf{Definition}: Ist $M$ eine Menge, so bezeichnet \\[0.3cm]
\hspace*{1.3cm} $2^M = \{ N \is N \subseteq M \}$ \\[0.3cm]
die \emph{Potenz-Menge} von $M$.
\vspace{0.5cm}

\textbf{Definition}: Ist $M$ endliche Menge, so bezeichnet \\[0.3cm]
\hspace*{1.3cm} $\mathtt{card}(M)$ \\[0.3cm]
die Anzahl der Elemente von $M$.

\textbf{Beispiele}:
\begin{enumerate}
\item $M = \emptyset$. \\[0.3cm]
      \hspace*{1.3cm} $2^M = \{ \emptyset \}$, \quad $\mathtt{card}(2^M) = 1$.
\item $M = \{1\}$. \\[0.3cm]
      \hspace*{1.3cm} $2^M = \left\{ \emptyset, \{1\} \rule{0pt}{16pt} \right\}$, \quad $\mathtt{card}(2^M) = 2$.
\item $M = \{1,2\}$ \\[0.3cm]
      \hspace*{1.3cm} $2^M = \left\{ \emptyset, \{1\}, \{2\}, \{1,2\} \rule{0pt}{16pt} \right\}$, \\[0.3cm]
      \hspace*{1.3cm} $\mathtt{card}(2^M) = 4$.
\end{enumerate}
\vspace{0.5cm}

\textbf{Satz}: Ist $M$ endlich, so gilt 
$$\mathtt{card}(2^M) = 2^{\mathtt{card}(M)}$$

\footnotesize

Mächtigkeit von Mengen
\end{center}
\vspace{0.5cm}

\footnotesize
\textbf{Definition}:  $M$ und $N$  \emph{gleichmächtig} g.d.w.
\begin{enumerate}
\item es existiert eine \underline{injektive} Funktion \\[0.3cm]
      \hspace*{1.3cm} $f: M \rightarrow N$ \quad und
\item es existiert eine \underline{injektive} Funktion \\[0.3cm]
      \hspace*{1.3cm} $g: N \rightarrow M$
\end{enumerate}
\textbf{Schreibweise}: \\[0.3cm]
\hspace*{1.3cm} $M \approx_{card} N$.

\textbf{Satz}: 
\begin{enumerate}
\item $M \approx_{card} M$
\item $K \approx_{card} M \wedge M \approx_{card} N \rightarrow K \approx_{card} N$
\end{enumerate}

\textbf{Beispiele}:
\begin{enumerate}
\item $\{1,2,3\} \approx_{card} \{a,b,c\}$
\item $\mathbb{N} \approx_{card} \mathbb{Z}$
\item $\mathbb{N} \approx_{card} \mathbb{N} \times \mathbb{N}$
\end{enumerate}

\begin{slide}{}
\normalsize
\begin{center}
Mächtigkeit von Mengen
\end{center}
\vspace{0.5cm}

\footnotesize
\textbf{Satz} (\textsl{Schröder-Bernstein})  Falls $M \approx_{card} N$, dann gibt es
eine \underline{bijektive} Funktion \\[0.3cm]
\hspace*{1.3cm} $h: M \rightarrow N$.
\vspace{0.5cm}


\textbf{Definition}: $N$ \emph{mächtiger} als $M$ g.d.w.
\begin{enumerate}
\item es existiert eine injektive Funktion \\[0.3cm]
      \hspace*{1.3cm} $f: M \rightarrow N$ \quad und
\item es existiert \underline{keine} injektive Funktion \\[0.3cm]
      \hspace*{1.3cm} $g: N \rightarrow M$.
\end{enumerate}
\textbf{Schreibweise}: $M \prec_{card} N$

\textbf{Beispiel}: $\{1,2,3\} \prec_{card} \{a,b,c,d\}$
\vspace{0.5cm}

\textbf{Satz}:  $K \prec_{card} M \wedge M \prec_{card} N \rightarrow K \prec_{card} N$
\vspace*{0.5cm}


\textbf{Definition}: Falls $M \approx_{card} \mathbb{N}$, so heißt $M$ \\
\emph{abzählbar unendlich}.

\textbf{Definition}: Falls $\mathbb{N} \prec_{card} M$, so heißt $M$ \\ 
\emph{überabzählbar}.

\textbf{Satz}: (Cantor) \\[0.3cm]
\hspace*{1.3cm} $\mathbb{N} \prec_{card} \mathbb{N}^\mathbb{N}$
\vspace{0.5cm}

\footnotesize
\textbf{Beweis}:
Annahme: $\mathbb{N} \approx_{card} \mathbb{N}^\mathbb{N}$

Schröder-Bernstein: existiert $f:\mathbb{N} \rightarrow \mathbb{N}^\mathbb{N}$ bijektiv.

Definiere\\[0.3cm]
\hspace*{1.3cm}  $\mathtt{diag}: \mathbb{N} \rightarrow \mathbb{N}$ \\[0.3cm]
\hspace*{1.3cm}  $\mathtt{diag}(n) := \left(\rule{0pt}{16pt}f(n)\right)(n) + 1$.\\[0.3cm]
$$ \mathtt{diag} \in \mathbb{N}^\mathbb{N} \quad 
      \mbox{also existiert}\; d \in \mathbb{N}\; \mbox{mit}\; f(d) = \mathtt{diag} $$
Konsequenz: 
$$ 
\begin{array}{lcl}
 \texttt{diag}(d) & = & \left(\rule{0pt}{16pt}f(d)\right)(d) + 1 \\
           & = & \texttt{diag}(d) + 1
\end{array}
$$        
Widerspruch!
      
\begin{center}
\framebox{\framebox{ $\;\displaystyle \mathbb{N}^\mathbb{N}$ ist überabzählbar.\ }}  
\end{center}

Dann gilt $\mathbb{P} \subseteq \mathbb{N}$.
