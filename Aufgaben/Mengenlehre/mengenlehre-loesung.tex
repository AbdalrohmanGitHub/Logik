\exercise 
Geben Sie die folgende Menge durch explizites Auflisten aller Elemente an: 
\\[0.2cm]
\hspace*{1.3cm}
$M := \{ x \el \mathbb{R} \mid x² - 4 \cdot x + 3 = 0 \}$.

\solution
Es gilt
\\[0.2cm]
\hspace*{1.3cm}
$x² - 4 \cdot x + 3 = (x - 1) \cdot (x - 3) = 0$ \quad g.d.w. \quad $x = 1 \vee x = 3$.
\\[0.2cm]
Also gilt 
\\[0.2cm]
\hspace*{1.3cm}
$M = \{1,3\}$. \qed
\vspace*{0.3cm}

\exercise
Geben Sie für die folgenden Mengen eine Definition in der Form 
\\[0.2cm]
\hspace*{1.3cm} 
$M := \{ g(x) \mid F(x) \}$
\\[0.2cm]
an. Dabei ist $g(x)$ ein Term und $F(x)$ ist eine logische Formel, die genau dann wahr ist,
wenn $x$ ein Element der Menge $M$ ist.  Beispiel:  Für die Menge 
\\[0.2cm]
\hspace*{1.3cm}
$M := \{ 2, 4, 8, \cdots \}$
\\[0.2cm]
lautet die Definition 
\\[0.2cm]
\hspace*{1.3cm}
$M := \{ 2^n \mid n \el \mathbb{N} \wedge n \geq 1 \}$.
\\[0.2cm]
Definieren Sie die folgenden Mengen in der oben gezeigten Form:
\begin{enumerate}
\item $M_1 = \{ 5, 10, 15, \cdots \}$
\item $M_2 = \{ 2, 4, 6, \cdots \} \cap \{ 3, 6, 9, \cdots \}$
\item $M_3 = \{ 2, 4, 6, \cdots \} \cup \{ 3, 6, 9, \cdots \}$
\item $M_4 = \{ 2, 4, 6, \cdots \} \;\backslash\; \{ 3, 6, 9, \cdots \}$
\end{enumerate}

\solution
\begin{enumerate}
\item $M_1 = \{ 5 \cdot n \mid n \el \mathbb{N} \wedge n \geq 1 \}$
\item $M_2 = \{ 6 \cdot n \mid n \el \mathbb{N} \wedge n \geq 1 \}$ 
\item $M_3 = \{ n  \mid n \el \mathbb{N} \;\wedge\;
              n \geq 1 \;\wedge\; (n \;\mathop{\mathtt{mod}}\; 2 = 0 \;\vee\; 
                                   n \;\mathop{\mathtt{mod}}\; 3 = 0)\}$
\item $M_4 = \{ 2 \cdot n \mid n \el \mathbb{N} \;\wedge\;
              n \geq 1 \;\wedge\; n \;\mathop{\mathtt{mod}}\; 3 \not= 0\}$ \qed
\end{enumerate}

\exercise
Es seien $M$ und $N$ Mengen und $R \subseteq M \times N$ sei eine binäre Relation.
Für eine Menge $X \subseteq M$ ist der Ausdruck $R(X)$ wie folgt definiert: 
\\[0.2cm]
\hspace*{1.3cm}
$R(X) := \{ y \el N \mid \exists x \el X: \pair(x,y) \el R \bigr\}$
\\[0.2cm]
Beweisen oder widerlegen Sie die folgenden Aussagen:
\begin{enumerate}
\item Für beliebe Teilmengen $A \subseteq M$ und $B \subseteq M$ gilt 
      \\[0.2cm]
      \hspace*{1.3cm}
      $R(A \cap B) = R(A) \cap R(B)$.
\item Für beliebe Teilmengen $A \subseteq M$ und $B \subseteq M$ gilt 
      \\[0.2cm]
      \hspace*{1.3cm}
      $R(A \cup B) = R(A) \cup R(B)$.
\item Falls Sie in Teil (a) oder (b) dieser Aufgabe zu dem Schluss kommen, dass
      die Aussage nicht allgemeingültig ist, dann überlegen Sie,
      welche Eigenschaften die Relation $R$ haben muss, damit 
      die Aussage richtig wird.
\end{enumerate}
\pagebreak

\solution
\begin{enumerate}
\item Die Aussage $R(A \cap B) = R(A) \cap R(B)$ ist im Allgemeinen falsch.
      Gegenbeispiel: 
      \\[0.2cm]
      \hspace*{1.3cm}
      $M = \{1,2\}$, \quad $N = \{3\}$, \quad $R = \{\pair(1,3), \pair(2,3)\}$,
      \quad $A = \{1\}$, \quad $B = \{2\}$.
      \\[0.2cm]
      Dann gilt 
      \\[0.2cm]
      \hspace*{1.3cm}
      $R(A) = \{3\}$, $R(B) = \{3\}$, \quad also \quad $R(A) \cap R(B) = \{3\}$, 
      \\[0.2cm]
      aber wir haben
      \\[0.2cm]
      \hspace*{1.3cm}
      $A \cap B = \emptyset$, \quad und folglich auch \quad  $R(A \cap B) = \emptyset$.
\item Die Aussage $R(A \cup B) = R(A) \cup R(B)$ ist richtig: 
      \\[0.2cm]
      \hspace*{1.3cm}
      $
      \begin{array}[t]{lcl}
              R(A \cup B) 
        & = & \{ y \el N \mid \exists x \el A \cup B: \pair(x,y) \el R \} 
              \\[0.2cm]
        & = & \{ y \el N \mid (\exists x \el A: \pair(x,y) \el R) \vee 
                              (\exists x \el B: \pair(x,y) \el R) \} 
              \\[0.2cm]
        & = & \{ y \el N \mid \exists x \el A: \pair(x,y) \el R \} \cup
              \{ y \el N \mid \exists x \el B: \pair(x,y) \el R \} 
              \\[0.2cm]
        & = & R(A) \cup R(B)
     \end{array}
      $
\item Das Gegenbeispiel für die Gleichung $R(A \cap B) = R(A) \cap R(B)$
      funktioniert nur deshalb, weil die Relation $R$ dort nicht links-eindeutig ist.
      Falls $R$ links-eindeutig ist, gilt 
      \\[0.2cm]
      \hspace*{1.3cm}
      $R(A \cap B) = R(A) \cap R(B)$
      \\[0.2cm]
      \textbf{Beweis}: Wir müssen zeigen, dass für alle $x \el M$ 
      \\[0.2cm]
      \hspace*{1.3cm}
      $y \el R(A \cap B)$ \quad g.d.w. \quad $y \el R(A) \cap R(B)$
      \\[0.2cm]
      gilt.  Wir nehmen zunächst $y \el R(A \cap B)$ an 
      (und zeigen $y \in R(A) \cap R(B)$). 
      Nach Definition von $R(A \cap B)$ heißt dass 
      \\[0.2cm]
      \hspace*{1.3cm}
      $\exists x \in A \cap B: \pair(x,y) \el R$
      \\[0.2cm]
      Da dieses $x$ dann sowohl in $A$ als auch in $B$ liegt,  folgt 
      \\[0.2cm]
      \hspace*{1.3cm}
      $(\exists x \el A: \pair(x,y) \el R) \wedge (\exists x \el B: \pair(x,y) \el R)$
      \\[0.2cm]
      Nach Definition von $R(X)$  heißt das 
      \\[0.2cm]
      \hspace*{1.3cm}
      $y \el R(A) \wedge y \el R(B)$, \quad also $y \in R(A) \cap R(B)$.
      \\[0.2cm]
      Wir nehmen nun umgekehrt an, dass $y \in R(A) \cap R(B)$ gilt.
      Also gilt 
      \\[0.2cm]
      \hspace*{1.3cm}
      $y \el R(A)$ \quad und \quad $y \el R(B)$.
      \\[0.2cm]
      Nach Definition von $R(X)$ heißt das 
      \\[0.2cm]
      \hspace*{1.3cm}
      $(\exists x_1 \el A : \pair(x_1,y) \el R) \;\wedge\; (\exists x_2 \el B : \pair(x_2,y) \el R)$.
      \\[0.2cm]
      An dieser Stelle kommt die Links-Eindeutigkeit zum tragen.  Wenn sowohl
      $\pair(x_1,y) \el R$ als auch $\pair(x_2,y) \el R$ gilt, dann muss $x_1 = x_2$ sein.
      Setzen wir $x := x_1$, so gilt dann 
      \\[0.2cm]
      \hspace*{1.3cm}
      $x \el A \wedge x \el B \wedge \pair(x,y) \el R$
      \\[0.2cm]
      Also haben wir 
      \\[0.2cm]
      \hspace*{1.3cm}
      $x \el A \cap B  \wedge \pair(x,y) \el R$.
      \\[0.2cm]
      Daraus folgt aber $y \el R(A \cap B)$. \qed
\end{enumerate}

\exercise
Beweisen oder widerlegen Sie die folgenden Gleichungen:
\begin{enumerate}
\item $(K \;\backslash\; M) \;\backslash\; N = K \;\backslash\; (M \cup N)$
\item $K \;\backslash\; (M \cap N) = (K \;\backslash\; M) \cap ( K \;\backslash\; N)$
\item $K \;\backslash\; (M \cup N) = (K \;\backslash\; M) \cup (K \;\backslash\; N)$
\item Falls Sie in Teil (b) oder (c) zu dem Schluss kommen,
      dass die Gleichung im Allgemeinen nicht gilt, dann geben Sie an,
      wie Sie die rechte Seite der Gleichung verändern müssen,
      damit die Gleichung allgemeingültig wird.
\end{enumerate}

\solution
\begin{enumerate}
\item Diese Gleichung ist allgemeingültig: 
      \\[0.2cm]
      \hspace*{1.3cm}
      $
      \begin{array}[t]{cl}
                      & x \el (K \;\backslash\; M) \;\backslash\; N \\[0.2cm]
      \Leftrightarrow & x \el (K \;\backslash\; M) \wedge \neg x \el N \\[0.2cm]
      \Leftrightarrow & x \el K \wedge \neg x \el M \wedge \neg x \el N \\[0.2cm]
      \Leftrightarrow & x \el K \wedge \neg (x \el M \vee x \el N) \\[0.2cm]
      \Leftrightarrow & x \el K \wedge \neg (x \el M \cup N) \\[0.2cm]
      \Leftrightarrow & x \el K \;\backslash\;  (M \cup N) 
      \end{array}
      $
\item Diese Gleichung ist im Allgemeinen falsch. Gegenbeispiel: 
      \\[0.2cm]
      \hspace*{1.3cm}
      $K = \{ 1, 2 \}$, $M = \{ 1\}$, $N = \{ 2 \}$.
      \\[0.2cm]
      Dann gilt einerseits
      \\[0.2cm]
      \hspace*{1.3cm}
      $
      \begin{array}[t]{cl}
            & K \;\backslash\;  (M \cap N) \\[0.2cm]
          = & \{1,2\} \;\backslash\; (\{1\} \cap \{2\}) \\[0.2cm]
          = & \{1,2\} \;\backslash\; \emptyset \\[0.2cm]
          = & \{1,2\} 
      \end{array}
      $ 
      \\[0.2cm]
      Andererseits haben wir
      \\[0.2cm]
      \hspace*{1.3cm}
      $
      \begin{array}[t]{cl}
            & (K \;\backslash\; M) \cap (K \;\backslash\; N) \\[0.2cm]          
          = & (\{1,2\} \;\backslash\; \{1\}) \cap (\{1,2\} \;\backslash\; \{2\}) \\[0.2cm]
          = & \{2\} \cap \{1\} \\[0.2cm]
          = & \emptyset
      \end{array}
      $
      \\[0.2cm]
      Also haben wir 
      \\[0.2cm]
      \hspace*{1.3cm}
      $K \;\backslash\;  (M \cap N) = \{1,2\} \not= \emptyset = (K \;\backslash\; M) \cap (K \;\backslash\; N)$.
\item Auch diese Gleichung ist im Allgemeinen falsch.
      Gegenbeispiel: 
      \\[0.2cm]
      \hspace*{1.3cm}
      $K = \{ 1, 2 \}$, $M = \{ 1\}$, $N = \{ 2 \}$.
      \\[0.2cm]
      Dann gilt einerseits
      \\[0.2cm]
      \hspace*{1.3cm}
      $
      \begin{array}[t]{cl}
            & K \;\backslash\;  (M \cup N) \\[0.2cm]
          = & \{1,2\} \;\backslash\; (\{1\} \cup \{2\}) \\[0.2cm]
          = & \{1,2\} \;\backslash\; \{1,2\} \\[0.2cm]
          = & \emptyset
      \end{array}
      $ 
      \\[0.2cm]
      Andererseits haben wir
      \\[0.2cm]
      \hspace*{1.3cm}
      $
      \begin{array}[t]{cl}
            & (K \;\backslash\; M) \cup (K \;\backslash\; N) \\[0.2cm]          
          = & (\{1,2\} \;\backslash\; \{1\}) \cup (\{1,2\} \;\backslash\; \{2\}) \\[0.2cm]
          = & \{2\} \cup \{1\} \\[0.2cm]
          = & \{1,2\}
      \end{array}
      $
      \\[0.2cm]
      Also haben wir 
      \\[0.2cm]
      \hspace*{1.3cm}
      $K \;\backslash\;  (M \cup N) = \emptyset \not= \{1,2\} = (K \;\backslash\; M) \cup (K \;\backslash\; N)$.
\item Die korrekte Form der Gleichungen lautet 
      \\[0.2cm]
      \hspace*{1.3cm}
      $K \;\backslash\; (M \cup N) = (K \;\backslash\; M) \cap (K \;\backslash\; N)$
      \quad und \\[0.2cm]
      \hspace*{1.3cm}
      $K \;\backslash\; (M \cap N) = (K \;\backslash\; M) \cup (K \;\backslash\; N)$.
      \\[0.2cm]
      Wir zeigen zunächst die erste dieser beiden Gleichungen:
      \\[0.2cm]
      \hspace*{1.3cm}
      $
      \begin{array}[t]{cl}
           & x \el K \ohne (M \cup N)           \\[0.2cm]
      \gdw & x \el K \wedge \neg x \el M \cup N \\[0.2cm]
      \gdw & x \el K \wedge \neg (x \el M \vee x \el N) \\[0.2cm]
      \gdw & x \el K \wedge \neg x \el M \wedge \neg x \el N \\[0.2cm]
      \gdw & (x \el K \wedge \neg x \el M) \wedge (x \el K \wedge \neg x \el N) \\[0.2cm]
      \gdw & (x \el K \ohne M) \wedge (x \el K \ohne N) \\[0.2cm]
      \gdw & x \el (K \ohne M) \cap (K \ohne N) 
      \end{array}
      $
      \\[0.2cm]
      Der Nachweis der zweiten Gleichung verläuft analog:
      \\[0.2cm]
      \hspace*{1.3cm}
      $
      \begin{array}[t]{cl}
           & x \el K \ohne (M \cap N)           \\[0.2cm]
      \gdw & x \el K \wedge \neg x \el M \cap N \\[0.2cm]
      \gdw & x \el K \wedge \neg (x \el M \wedge x \el N) \\[0.2cm]
      \gdw & x \el K \wedge (\neg x \el M \vee \neg x \el N) \\[0.2cm]
      \gdw & (x \el K \wedge \neg x \el M) \vee (x \el K \wedge \neg x \el N) \\[0.2cm]
      \gdw & (x \el K \ohne M) \vee (x \el K \ohne N) \\[0.2cm]
      \gdw & x \el (K \ohne M) \cup (K \ohne N) 
      \end{array}
      $
\end{enumerate}

\exercise 
Für eine endliche Menge $M$ bezeichnet $\textsl{card}(M)$ die Anzahl der Elemente.
\begin{enumerate}
\item Geben Sie eine Formel zur Berechnung von $\textsl{card}(M \times M)$ an.
\item Zeigen Sie, dass für die Potenz-Menge $2^M$ gilt 
      \\[0.2cm]
      \hspace*{1.3cm}
      $\textsl{card}\bigl(2^M\bigr) = 2^{\textsl{card}(M)}$.
\item Zeigen Sie, dass die Gleichung 
      \\[0.2cm]
      \hspace*{1.3cm}
      $\textsl{card}(A \cup B) = \textsl{card}(A) + \textsl{card}(B)$
      \\[0.2cm]
      im Allgemeinen falsch ist.
\item Können Sie die rechte Seite der Gleichung in Teil (c) so ändern,
      dass die Gleichung richtig wird?
\item Geben Sie eine Formel zur Berechnung von $\textsl{card}(A \cup B \cup C)$ an.
\end{enumerate}

\solution
\begin{enumerate}
\item Es gilt $\textsl{card}(M \times N) = \textsl{card}(M) \cdot \textsl{card}(N)$.
\item Wir definieren $n := \textsl{card}(M)$ und zeigen die Behauptung
      durch Induktion nach der Zahl $n$ der Elemente der Menge $M$.
      \begin{description}
      \item[I.A.] $n = 0$: Dann gilt $M= \emptyset$.
                  Wir haben
                  \\[0.2cm]
                  \hspace*{1.3cm}
                  $2^M = \{\emptyset\}$
                  \\[0.2cm]
                  Also gilt 
                  \\[0.2cm]
                  \hspace*{1.3cm}
                  $\textsl{card}(2^M) = \textsl{card}(\{\emptyset\}) = 1 = 2^0 = 2^{\textsl{card}(M)}$
                  \\[0.2cm]
                  und damit stimmt die Behauptung in diesem Fall.
      \item[I.S.] $n \mapsto n + 1$: Dann hat $M$ die Form 
                  \\[0.2cm]
                  \hspace*{1.3cm}
                  $M = K \cup \{x\}$ \quad mit $\textsl{card} (K) = n$.
                  \\[0.2cm]
                  Die Potenzmenge von M kann dann aus der Potenzmenge
                  von $K$ wie folgt gebildet werden: 
                  \\[0.2cm]
                  \hspace*{1.3cm}
                  $2^M = 2^K \cup \bigl\{ A \cup \{x\} \mid A \el 2^K \bigr\}$
                  \\[0.2cm]
                  Damit sehen wir, dass die Potenzmenge von $M$ aus zwei Teilen besteht:
                  der erste Teil ist mit der Potenzmenge von $K$ identisch, den zweiten
                  Teil erhalten wir, wenn wir zu jeder Menge aus der Potenzmenge von $K$
                  noch das Element $x$ hinzufügen.  Offenbar sind diese beiden Mengen
                  einerseits gleichgroß und andererseits disjunkt.  Daher gilt
                  \\[0.2cm]
                  \hspace*{1.3cm}
                  $\textsl{card}(2^M) = \textsl{card}(2^K) + \textsl{card}(2^K)  
                   = 2 \cdot \textsl{card}(2^K) \stackrel{IV}{=} 2 \cdot 2^n = 2^{n+1}$.
      \end{description}
\item Wir erhalten ein triviales Gegenbeispiel, wenn wir $A= \{1\} = B$ setzen, denn dann
      gilt 
      \\[0.2cm]
      \hspace*{1.3cm}
      $\textsl{card}(A \cup B) = \textsl{card}(\{1\}) = 1 \not = 1 + 1 = \textsl{card}(A) + \textsl{card}(B)$.
\item Das Problem ist, dass in der Summe $\textsl{card}(A) + \textsl{card}(B)$ 
      die Elemente, die sowohl in $A$ als auch in $B$ vorkommen, doppelt gezählt werden.
      Das sind aber genau die Elemente aus dem Schnitt von $A$ und $B$.  Die
      Anzahl dieser Elemente muss also abgezogen werden.  Damit lautet die korrekte 
      Formel 
      \\[0.2cm]
      \hspace*{1.3cm}
      $\textsl{card}(A \cup B) = \textsl{card}(A) + \textsl{card}(B) - \textsl{card}(A \cap B)$.
\item Die Formel lautet 
      \\[0.2cm]
      \hspace*{1.3cm}
      $
      \begin{array}[t]{lcl}
        \textsl{card}(A \cup B \cup C) 
       & = & \;\;\;\textsl{card}(A) + \textsl{card}(B) + \textsl{card}(B) \\[0.2cm]
       &   & - \;\textsl{card}(A \cap B) - \textsl{card}(A \cap B) - \textsl{card}(A \cap B) \\[0.2cm]
       &   & + \;\textsl{card}(A \cap B \cap C)
      \end{array}
      $
      \\[0.2cm]
      Entscheidend ist hier der Term $\textsl{card}(A \cap B \cap C)$.
      Er sorgt dafür, dass Elemente, die gleichzeitig in $A$, $B$ und $C$
      auftreten, korrekt gezählt werden. \qed
\end{enumerate}

\exercise
Auf der Menge der natürlichen Zahlen werde die Relation $R$ als 
\\[0.2cm]
\hspace*{1.3cm}
$R := \bigl\{ \pair(n, 2\cdot n) \mid n \el \mathbb{N} \bigr\}$
\\[0.2cm]
definiert.  Berechnen Sie den transitiven Abschluss der Relation $R$.
\pagebreak

\solution
Wir zeigen zunächst durch Induktion über $k \el \mathbb{N}$, dass für alle positiven natürlichen Zahlen $k$ folgendes gilt:
\\[0.2cm]
\hspace*{1.3cm}
$R^k = \bigl\{ \pair(n, 2^k\cdot n) \mid n \el \mathbb{N} \bigr\}$ 
\begin{enumerate}
\item[I.A.] $k = 1$:  Wegen 
            \\[0.2cm]
            \hspace*{1.3cm}
            $R^1 = R = \bigl\{ \pair(n, 2\cdot n) \mid n \el \mathbb{N} \bigr\} = 
                       \bigl\{ \pair(n, 2^1\cdot n) \mid n \el \mathbb{N} \bigr\}$ 
            \\[0.2cm]
            ist die Behauptung in diesem Fall offensichtlich.
\item[I.S.] $k \mapsto k + 1$:  In diesem Fall haben wir 
            \\[0.2cm]
            \hspace*{1.3cm}
            $
            \begin{array}[t]{lcl}
                     R^{k+1} 
               & = & R \circ R^k
                     \\[0.2cm]
& \stackrel{IV}{=} & R \circ \bigl\{ \pair(n, 2^k \cdot n) \mid n \el \mathbb{N} \bigr\} 
                     \\[0.2cm]
               & = & \bigl\{ \pair(m, 2   \cdot m) \mid m \el \mathbb{N} \bigr\} \circ 
                     \bigl\{ \pair(n, 2^k \cdot n) \mid n \el \mathbb{N} \bigr\} 
                     \\[0.2cm]
               & = & \bigl\{ \langle m, 2^k \cdot (2 \cdot m)\rangle \mid n \el \mathbb{N} \bigr\} 
                     \\[0.2cm]
               & = & \bigl\{ \pair(m, 2^{k+1} \cdot m) \mid n \el \mathbb{N} \bigr\} 
            \end{array}
            $
\end{enumerate}
Wegen 
\\[0.2cm]
\hspace*{1.3cm}
$R^+ = \bigcup\limits_{k=1}^\infty R^k$
\\[0.2cm]
 hat die Relation $R^+$ dann die Form 
\\[0.2cm]
\hspace*{1.3cm}
$R^+ = \bigl\{ \pair(n, 2^k\cdot n) \mid n \el \mathbb{N} \wedge k \el \mathbb{N} \wedge k \geq 1 \bigr\}$. \qed
\vspace*{0.3cm}

\exercise
Es sei $M$ eine endliche Menge und 
\\[0.2cm]
\hspace*{1.3cm}
$\Gamma: 2^M \rightarrow 2^M$
\\[0.2cm]
sei eine Funktion, die Teilmengen von $M$ in Teilmengen von $M$ abbildet.
Eine Funktion, die Mengen in Mengen abbildet, wird als \emph{Operator} bezeichnet.
Wir nennen den Operator \emph{monoton}, wenn 
\\[0.2cm]
\hspace*{1.3cm}
$\forall A,B \el 2^M: A \subseteq B \rightarrow \Gamma(A) \subseteq \Gamma(B)$
\\[0.2cm]
gilt.  Wir definieren induktiv eine Folge $(X_n)_{n\el\mathbb{N}}$ von Teilmengen der
Menge $M$ wie folgt:
\renewcommand{\labelenumi}{\arabic{enumi}.}
\begin{enumerate}
\item Induktions-Anfang: $n=0$
      \\[0.2cm]
      \hspace*{1.3cm}
      $X_0 := \emptyset$.
\item Induktions-Schritt: $n \mapsto n+1$
      \\[0.2cm]
      \hspace*{1.3cm}
      $X_{n+1} := \Gamma(X_n)$
\end{enumerate}
\renewcommand{\labelenumi}{(\alph{enumi})}
Nehmen Sie im Folgenden an, dass $\Gamma$ monoton ist und bearbeiten Sie die folgenden Teilaufgaben.
\begin{enumerate}
\item Zeigen Sie durch vollständige Induktion 
      \\[0.2cm]
      \hspace*{1.3cm}
      $\forall n \el \mathbb{N}: X_n \subseteq X_{n+1}$.
\item Es gibt ein $k\el \mathbb{N}$, so dass gilt:
      $\forall n \el \mathbb{N}: n \geq k \rightarrow X_n = X_k$

      Die Menge $X_k$ definieren wir dann als den \emph{Grenzwert} der Folge $(X_n)_{n\el\mathbb{N}}$:
      \\[0.2cm]
      \hspace*{1.3cm}
      $\lim\limits_{n\rightarrow\infty} X_n := X_k$.
\item Zeigen Sie, dass es eine Menge $X \subseteq M$ gibt, so dass
      $\Gamma(X) = X$ 
      gilt. 

      Eine Menge $Y$ mit der Eigenschaft $\Gamma(Y) = Y$ heißt \emph{Fixpunkt} des Operators $\Gamma$.
\item Ist $Y$ ein Fixpunkt des Operators $\Gamma$, so gilt 
      \\[0.2cm]
      \hspace*{1.3cm}
      $\lim\limits_{n\rightarrow\infty} X_n \subseteq Y$.
\end{enumerate}
\textbf{Bemerkung}: 
Die Aufgabe zeigt, dass auf einer endlichen Menge $M$ jeder monotone Operator 
\\[0.2cm]
\hspace*{1.3cm}
$\Gamma: 2^M \rightarrow 2^M$
\\[0.2cm]
einen kleinsten Fixpunkt hat und gibt darüber hinaus ein Verfahren an, mit dem dieser Fixpunkt 
berechnet werden kann. 

\solution
\begin{enumerate}
\item Wir beweisen die Behauptung durch Induktion $n$.
      \begin{enumerate}
      \item[I.A.] $n = 0$:  Es gilt 
                  \\[0.2cm]
                  \hspace*{1.3cm}
                  $
                  \begin{array}[t]{lcl}
                                  & X_0 \subseteq X_1 \\[0.2cm]
                  \Leftrightarrow & \emptyset \subseteq X_1
                  \end{array}
                  $
                  \\[0.2cm] 
                  Die letzte Behauptung ist offenbar richtig.
      \item[I.S.] $n \mapsto n + 1$:  Zunächst wissen wir nach Induktions-Voraussetzung, dass
                  \\[0.2cm]
                  \hspace*{1.3cm}
                  $X_n \subseteq X_{n+1}$
                  \\[0.2cm]
                  gilt.  Aus der Monotonie des Operators $\Gamma$ folgt dann 
                  \\[0.2cm]
                  \hspace*{1.3cm}
                  $\Gamma(X_n) \subseteq \Gamma(X_{n+1})$
                  \\[0.2cm]
                  Nach der Definition der Folge $(X_n)_{n\el\mathbb{N}}$ ist die
                  letzte Ungleichung äquivalent zu 
                  \\[0.2cm]
                  \hspace*{1.3cm}
                  $X_{n+1} \subseteq X_{n+2}$
                  \\[0.2cm]
                  und das ist genau die Behauptung für $n+1$.
      \end{enumerate}
\item Nach Teil (a) wissen wir 
      \\[0.2cm]
      \hspace*{1.3cm}
      $X_0 \subseteq X_1 \subseteq X_2 \subseteq \cdots \subseteq X_n \subseteq X_{n+1} \subseteq 
       \cdots \subseteq M$.
      \\[0.2cm]
      Da aus $A \subseteq B$ offenbar $\textsl{card(A)} \leq \textsl{card}(B)$ folgt,
      heißt das 
      \\[0.2cm]
      \hspace*{0.8cm}
      $\textsl{card}(X_0) \leq \textsl{card}(X_1) \leq \textsl{card}(X_2) \leq \cdots \leq 
      \textsl{card}(X_n) \leq \textsl{card}(X_{n+1}) \leq \cdots \leq \textsl{card}(M)$
      \\[0.2cm]
      Da $M$ eine endliche Menge ist, kann nicht in allen Ungleichungen 
      \\[0.2cm]
      \hspace*{1.3cm}
      $\textsl{card}(X_n) \leq \textsl{card}(X_{n+1})$
      \\[0.2cm]
      ein echtes Ungleichheitszeichen gelten. Wir definieren 
      \\[0.2cm]
      \hspace*{1.3cm}
      $k := \min \bigl\{ n \el \mathbb{N} \mid \textsl{card}(X_n) = \textsl{card}(X_{n+1}) \bigr\}$
      \\[0.2cm]
      Da nun einerseits $\textsl{card}(X_k) = \textsl{card}(X_{k+1})$ und andererseits
      $X_k \subseteq X_{k+1}$ gilt, haben wir sogar
      \\[0.2cm]
      \hspace*{1.3cm}
      $X_k = X_{k+1} = \Gamma(X_k)$.
      \\[0.2cm]
      Dann zeigt eine triviale Induktion, dass für alle $n \geq k$ ebenfalls
      \\[0.2cm]
      \hspace*{1.3cm}
      $X_n = X_k$
      \\[0.2cm]
      gilt.
\item Nach Teil (b) haben wir
      \\[0.2cm]
      \hspace*{1.3cm}
      $\Gamma(X_k) = X_{k+1} = X_k$.
      \\[0.2cm]
      Damit ist $X_k$ der gesuchte Fixpunkt.
\item Wir zeigen durch Induktion über $n$, dass für alle $n \el \mathbb{N}$ 
      \\[0.2cm]
      \hspace*{1.3cm}
      $X_n \subseteq Y$ 
      \\[0.2cm]
      gilt. Daraus folgt dann die Behauptung, denn es gilt ja 
      $\lim\limits_{n\rightarrow\infty} X_n = \bigcup\limits_{n=1}^\infty X_n$.
      \begin{enumerate}
      \item[I.A.] $n = 0$:  Wegen $X_0 = \emptyset$ ist dieser Fall trivial.
      \item[I.S.] $n \mapsto n + 1$: Nach Induktions-Voraussetzung wissen wir 
                  \\[0.2cm]
                  \hspace*{1.3cm}
                  $X_n \subseteq Y$.
                  \\[0.2cm]
                  Aufgrund der Monotonie des Operators $\Gamma$ folgt daraus 
                  \\[0.2cm]
                  \hspace*{1.3cm}
                  $\Gamma(X_n) \subseteq \Gamma(Y)$.
                  \\[0.2cm]
                  Nach Definition der Folge $(X_n)_{n\el\mathbb{N}}$ gilt
                  $\Gamma(X_n) = X_{n+1}$ und da $Y$ ein Fixpunkt von $\Gamma$ ist,
                  gilt $\Gamma(Y) = Y$.  Also haben wir 
                  \\[0.2cm]
                  \hspace*{1.3cm}
                  $X_{n+1} \subseteq Y$
                  \\[0.2cm]
                  und das war zu zeigen. \qed
      \end{enumerate}
\end{enumerate}
